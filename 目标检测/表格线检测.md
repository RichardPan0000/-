

# 表格线检测



``` 
1.灰度化之后，全二值化。能够比只灰度化，得到更多的直线。
2.同时不用canny算子去进行强化边缘。（canny算子进行边缘连接和非极大值一致，生成细而连续的边缘）。
3.对二值图进行膨胀和闭操作，增强横竖线
```



当然一般需要透视变换，将原始图还原出来，然后剪切调无关部分（或者mask掉无关部分）。





霍夫变换

import cv2
import math
import numpy as np
import os
import time

# 创建输出目录
output_dir = "output_images"
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

def resize_image(image, target_size=(640, 640)):
    """
    将图像调整为指定大小，保持宽高比
    """
    h, w = image.shape[:2]
    # 计算缩放比例
    scale = min(target_size[0]/w, target_size[1]/h)
    # 计算新的尺寸
    new_w = int(w * scale)
    new_h = int(h * scale)
    # 调整图像大小
    resized = cv2.resize(image, (new_w, new_h), interpolation=cv2.INTER_AREA)
    # 创建目标大小的黑色背景
    result = np.zeros((target_size[1], target_size[0], 3), dtype=np.uint8)
    # 将调整后的图像放在中心位置
    y_offset = (target_size[1] - new_h) // 2
    x_offset = (target_size[0] - new_w) // 2
    result[y_offset:y_offset+new_h, x_offset:x_offset+new_w] = resized
    return result

def detect_lines_and_save():

    time1=time.time()
    # 读取图像
    # image = cv2.imread("./fig_match/qr_large.jpg")
    image = cv2.imread("line.jpg")
    # image = cv2.imread("line3.jpg") # 使用用户指定的图片路径
    # image = cv2.imread("line2.jpg") # 使用用户指定的图片路径
    if image is None:
        # 如果图像未找到，打印错误信息并退出
        print("Error: 图像文件未找到。请检查文件路径。")
        exit()

    # 调整图像大小为640x640
    image = resize_image(image, (1280, 1280))
    
    # 保存原始图片
    cv2.imwrite(os.path.join(output_dir, "original.jpg"), image)

    # 获取原始图像的尺寸，用于计算直线长度阈值
    original_h, original_w = image.shape[:2]
    # 计算图像长宽中的较小者
    min_dim = min(original_h, original_w)

    # 设置直线长度阈值的百分比
    # line_length_threshold_percentage = 0.05 # 例如，图像较小尺寸的5%
    line_length_threshold_percentage = 0.40 # 例如，图像较小尺寸的5%
    # 计算实际的直线长度阈值
    line_length_threshold = min_dim * line_length_threshold_percentage
    print(f"直线长度阈值 (小于图像较小尺寸的 {line_length_threshold_percentage*100:.2f}%) 为: {line_length_threshold:.2f} 像素")

    # 转换为灰度图
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    # 设置阈值，二值化处理
    black_thr = 150
    ret, binary = cv2.threshold(gray, black_thr, 255, cv2.THRESH_BINARY)

    # 保存二值化图片
    cv2.imwrite(os.path.join(output_dir, "binary.jpg"), binary)

    # 查找轮廓
    # cv2.findContours 在 OpenCV 4.x 版本中返回两个值
    contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if not contours:
        print("Error: 未检测到任何轮廓。")
        exit()

    # 找到最大轮廓
    max_contour = max(contours, key=cv2.contourArea)

    # 计算包围矩形框
    x, y, w, h = cv2.boundingRect(max_contour)
    paper_region = image[y:y + h, x:x + w]

    # 检查裁剪区域是否有效
    if paper_region.size == 0:
        print("Error: 裁剪区域为空。")
        exit()

    # 保存裁剪后的区域
    cv2.imwrite(os.path.join(output_dir, "cropped_region.jpg"), paper_region)

    # 预处理裁剪区域
    gray_region = cv2.cvtColor(paper_region, cv2.COLOR_BGR2GRAY)
    ret, binary_region = cv2.threshold(gray_region, black_thr, 255, cv2.THRESH_BINARY)

    # 保存裁剪区域的二值化图片
    cv2.imwrite(os.path.join(output_dir, "cropped_binary.jpg"), binary_region)

    # 设置结构核
    dh, dw = binary_region.shape
    # 调整膨胀核大小的计算方式，避免过大或过小
    hors_k = max(1, int(math.sqrt(dw) * 0.1))  # 横向膨胀核大小
    vert_k = max(1, int(math.sqrt(dh) * 0.1))  # 纵向膨胀核大小
    # 适当增加迭代次数以确保连接
    dilate_iterations = 3

    kernel_h = cv2.getStructuringElement(cv2.MORPH_RECT, (hors_k, 1))
    # 对二值图进行膨胀，然后取反得到黑色的横线
    hors = ~cv2.dilate(binary_region, kernel_h, iterations=dilate_iterations)

    kernel_v = cv2.getStructuringElement(cv2.MORPH_RECT, (1, vert_k))
    # 对二值图进行膨胀，然后取反得到黑色的竖线
    verts = ~cv2.dilate(binary_region, kernel_v, iterations=dilate_iterations)

    # 使用闭操作补充不完整的线条（膨胀和腐蚀）
    # 先合并横线和竖线检测结果
    merged_lines = cv2.bitwise_or(hors, verts)
    kernel_close = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))  # 适当调整核的大小
    closed = cv2.morphologyEx(merged_lines, cv2.MORPH_CLOSE, kernel_close)
    borders = closed

    # 保存形态学处理后的图片
    cv2.imwrite(os.path.join(output_dir, "morphological.jpg"), borders)

    # 霍夫直线变换
    # 调整霍夫变换参数，特别是 threshold 和 minLineLength
    # detected_lines = cv2.HoughLinesP(borders, 1, np.pi / 180, threshold=50, minLineLength=50, maxLineGap=10)
    # detected_lines = cv2.HoughLinesP(borders, 1, np.pi / 360, threshold=50, minLineLength=50, maxLineGap=40)
    detected_lines = cv2.HoughLinesP(borders, 1, np.pi / 360, threshold=50, minLineLength=line_length_threshold, maxLineGap=40)
    if detected_lines is None:
        print("No lines detected.")
        # 即使没有检测到直线，也显示原始图像，并保存
        output_image = image.copy()
        cv2.imwrite(os.path.join(output_dir, "final_result.jpg"), output_image)
        cv2.imshow("Detected Grid with Cells", output_image)
        cv2.waitKey(0)
        cv2.destroyAllWindows()
        exit()

    # 筛选符合长度阈值的直线
    filtered_lines = []
    for line in detected_lines:
        x1, y1, x2, y2 = line[0]
        # 计算直线长度
        line_length = math.sqrt((x2 - x1)**2 + (y2 - y1)**2)
        if line_length >= line_length_threshold:
            filtered_lines.append(line)

    print(f"检测到 {len(detected_lines)} 条直线。")
    print(f"筛选后，有 {len(filtered_lines)} 条直线长度超过阈值。")

    # 创建副本以绘图
    output_image = image.copy()

    # 绘制筛选后的直线
    for line in filtered_lines:
        x1, y1, x2, y2 = line[0]
        # 将直线坐标转换回原图位置
        cv2.line(output_image, (x1 + x, y1 + y), (x2 + x, y2 + y), (0, 255, 0), 2)

    time2=time.time()
    time_spend=time2-time1
    print('处理时间为%s',time_spend)
    # 保存最终结果图片
    cv2.imwrite(os.path.join(output_dir, "final_result.jpg"), output_image)
