```python
import torch
import torch.nn as nn
from torchvision.ops import DeformConv2d

class MyDCNBlock(nn.Module):
    def __init__(self, in_ch, out_ch, k=3, padding=1):
        super().__init__()
        self.offset_conv = nn.Conv2d(in_ch, 2*k*k, kernel_size=3, padding=1)
        self.mask_conv   = nn.Conv2d(in_ch, k*k,     kernel_size=3, padding=1)
        self.dcn         = DeformConv2d(in_ch, out_ch, kernel_size=k, padding=padding)

    def forward(self, x):
        offset = self.offset_conv(x)
        mask   = torch.sigmoid(self.mask_conv(x))
        return self.dcn(x, offset, mask)

    
    
 # 使用测试
self.conv3 = MyDCNBlock(128, 256)

x = torch.randn(1, 128, 64, 64)
y = self.conv3(x)
print(y.shape)  # 应输出 [1, 256, 64, 64]

```



偏移通道数为 `2*K*K`，mask 通道数为 `K*K`

如果 `mask` 不为 None，则执行 DCNv2，否则为 DCNv1
